\newcommand{\coursedesc}{Introduction to Web development (design and creation) using current standards for client-side content delivery (e.g., XHTML and CSS). Students will learn to create and publish a multi-page, static-content website using associated applications. Special focus is given to user interface design, data presentation, and data organization.}

\newcommand{\coursecredithours}{3}
\newcommand{\courseprereq}{CIS 225, 236 (with a grade of C or higher in both).}

\newcommand{\courseobjectives}{
	% [COB Goals 2,3]
	\item Interperet modern web technologies and use browser development tools to dissect web pages.
	% [COB Goal 3]
	\item Design, create, and modify web pages constructed in HTML and CSS and basic JavaScript. 
	\item Analyze and compare user interfaces and discuss accessibility concerns.
	\item Discuss privacy, security, and ethical concerns of data gathering/mining as related to web design.
}

	 \item Web Browsers
	 \item User Interface Design
	 \item Accessibility
	 \item Privacy and Security
	 \item Ethics and Social Responsibility
	 \item Information Technology and E-Commerce
	\end{itemize}
	\item System and network auditing
	\item System and network hardening
}
